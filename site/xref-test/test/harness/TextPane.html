6   package test.harness;
7   
8   import io.github.olyutorskii.quetexj.MaxTracker;
9   import io.github.olyutorskii.quetexj.MvcFacade;
10  import javax.swing.Box;
11  import javax.swing.BoxLayout;
12  import javax.swing.JButton;
13  import javax.swing.JCheckBox;
14  import javax.swing.JComponent;
15  import javax.swing.JPanel;
16  import javax.swing.JScrollBar;
17  import javax.swing.JScrollPane;
18  import javax.swing.JTextArea;
19  
20  /**
21   * Pseudo endless text component.
22   */
23  @SuppressWarnings("serial")
24  class TextPane extends JComponent{
25  
26      private final JScrollPane scrollPane;
27      private final JButton clearBtn;
28      private final JCheckBox trackBtn;
29  
30  
31      /**
32       * Constructor.
33       *
34       * @param facade MVC
35       */
36      TextPane(MvcFacade facade){
37          super();
38  
39          JTextArea textArea = facade.getTextArea();
40          this.scrollPane = new JScrollPane();
41          this.scrollPane.setViewportView(textArea);
42  
43          JScrollBar vertBar = this.scrollPane.getVerticalScrollBar();
44          vertBar.setModel(facade.getVerticalBoundedRangeModel());
45  
46          this.clearBtn = new JButton();
47          this.clearBtn.setAction(facade.getClearAction());
48          this.clearBtn.setText("clear");
49  
50          this.trackBtn = new JCheckBox("tracking last");
51          this.trackBtn.setModel(facade.getTrackSwitchButtonModel());
52  
53          MaxTracker maxTracker = facade.getMaxTracker();
54          maxTracker.setTrackingMode(true);
55  
56          design();
57  
58          return;
59      }
60  
61  
62      /**
63       * design layout.
64       */
65      private void design(){
66          BoxLayout layout;
67  
68          JPanel pane = new JPanel();
69          layout = new BoxLayout(pane, BoxLayout.LINE_AXIS);
70          pane.setLayout(layout);
71          pane.add(this.scrollPane);
72          pane.add(Box.createVerticalGlue());
73  
74          JPanel buttons = new JPanel();
75          layout = new BoxLayout(buttons, BoxLayout.LINE_AXIS);
76          buttons.setLayout(layout);
77          buttons.add(this.clearBtn);
78          buttons.add(Box.createHorizontalGlue());
79          buttons.add(this.trackBtn);
80  
81          layout = new BoxLayout(this, BoxLayout.PAGE_AXIS);
82          setLayout(layout);
83          add(pane);
84          add(buttons);
85  
86          return;
87      }
88  
89  }
