28  /**
29   * Timer controll GUI.
30   */
31  @SuppressWarnings("serial")
32  class TimerPanel extends JComponent{
33  
34      private final Timer timer;
35  
36      private final JComponent intervalPanel;
37  
38      private final JComponent counterPanel;
39      private final JToggleButton onOffButton;
40      private final JLabel eventNumLabel;
41      private int eventNum;
42  
43      private final Map<String, Integer> msecCommand = new HashMap<>();
44  
45  
46      /**
47       * Constructor.
48       *
49       * @param ticks interval infos
50       */
51      TimerPanel(IntervalTick... ticks){
52          super();
53  
54          this.eventNumLabel = new JLabel();
55          this.eventNum = 0;
56          updateEventCounter();
57  
58          this.onOffButton = new JToggleButton();
59          updateOnOffLabel();
60          this.onOffButton.addItemListener(ev -> {
61              eventOnOffSwitch();
62          });
63  
64          this.intervalPanel = buildIntervalPanel(ticks);
65          this.counterPanel = buildCounterPanel();
66  
67          design();
68  
69          for(IntervalTick tick : ticks){
70              this.msecCommand.put(tick.label, tick.msec);
71          }
72  
73          this.timer = buildTimer();
74          this.timer.addActionListener(ev -> {
75              eventCounterIncrement();
76          });
77  
78          return;
79      }
80  
81  
82      /**
83       * Build timer.
84       *
85       * @return timer
86       */
87      private static Timer buildTimer(){
88          Timer result = new Timer(0, null);
89  
90          result.setInitialDelay(1);
91  
92          result.setCoalesce(true);
93          result.setRepeats(true);
94  
95          return result;
96      }
97  
98  
99      /**
100      * Build interval radio button panel.
101      *
102      * @param ticks interval definition
103      * @return interval radio button panel
104      */
105     private JComponent buildIntervalPanel(IntervalTick... ticks){
106         JRadioButton defaultInterval = null;
107         ButtonGroup buttonGroup = new ButtonGroup();
108 
109         JComponent panel = Box.createVerticalBox();
110 
111         for(IntervalTick tick : ticks){
112             JRadioButton radioButton = new JRadioButton();
113 
114             String label = tick.label;
115             radioButton.setText(label);
116             radioButton.setActionCommand(label);
117 
118             radioButton.addActionListener(ev -> {
119                 eventIntervalChange(ev);
120             });
121 
122             buttonGroup.add(radioButton);
123             panel.add(radioButton);
124 
125             if(tick.isDefault){
126                 defaultInterval = radioButton;
127             }
128         }
129 
130         if(defaultInterval != null){
131             EventQueue.invokeLater(defaultInterval::doClick);
132         }
133 
134         return panel;
135     }
136 
137     /**
138      * Build timer event counter panel.
139      *
140      * @return timer event counter panel
141      */
142     private JComponent buildCounterPanel(){
143         Box number = Box.createVerticalBox();
144         this.eventNumLabel.setAlignmentX(Component.RIGHT_ALIGNMENT);
145         number.add(this.eventNumLabel);
146 
147         JComponent cardComp = new JPanel();
148         cardComp.setLayout(new CardLayout());
149         cardComp.add(number);
150         cardComp.add(new JLabel("99999"));
151 
152         JComponent result;
153         result = Box.createHorizontalBox();
154 
155         result.add(new JLabel("Event #:"));
156         result.add(cardComp);
157 
158         return result;
159     }
160 
161     /**
162      * Design layout.
163      */
164     private void design(){
165         GridBagLayout layout = new GridBagLayout();
166         GridBagConstraints constraints = new GridBagConstraints();
167         setLayout(layout);
168 
169         Border border = new TitledBorder("");
170         this.intervalPanel.setBorder(border);
171 
172         constraints.gridheight = GridBagConstraints.REMAINDER;
173         add(this.intervalPanel, constraints);
174 
175         constraints.gridwidth = GridBagConstraints.REMAINDER;
176         constraints.gridheight = 1;
177         constraints.anchor = GridBagConstraints.CENTER;
178         constraints.fill = GridBagConstraints.HORIZONTAL;
179         constraints.weighty = 0.5;
180         constraints.insets = new Insets(3, 3, 3, 3);
181         add(this.onOffButton, constraints);
182 
183         add(this.counterPanel, constraints);
184 
185         EventQueue.invokeLater(() -> {
186             this.onOffButton.requestFocusInWindow();
187         });
188 
189         return;
190     }
191 
192     /**
193      * Receive event counter increment event.
194      */
195     private void eventCounterIncrement(){
196         this.eventNum++;
197         updateEventCounter();
198         return;
199     }
200 
201     /**
202      * Update event counter label.
203      */
204     private void updateEventCounter(){
205         String label = Integer.toString(this.eventNum);
206         this.eventNumLabel.setText(label);
207         return;
208     }
209 
210     /**
211      * Receive START/STOP button event.
212      */
213     private void eventOnOffSwitch(){
214         boolean isSelected = updateOnOffLabel();
215 
216         if(isSelected) this.timer.start();
217         else           this.timer.stop();
218 
219         return;
220     }
221 
222     /**
223      * Update START/STOP button label.
224      *
225      * @return true if selected
226      */
227     private boolean updateOnOffLabel(){
228         String label;
229         boolean isSelected = this.onOffButton.isSelected();
230 
231         if(isSelected) label = "STOP";
232         else           label = "START";
233 
234         this.onOffButton.setText(label);
235 
236         return isSelected;
237     }
238 
239     /**
240      * Receive interval change event.
241      *
242      * @param ev action event
243      */
244     private void eventIntervalChange(ActionEvent ev){
245         int msec = parseActionEvent(ev);
246         this.timer.setDelay(msec);
247         if(this.timer.isRunning()){
248             this.timer.restart();
249         }
250         return;
251     }
252 
253     /**
254      * Parse action event to milli sec interval value.
255      *
256      * @param ev action event
257      * @return milli sec interval
258      */
259     private int parseActionEvent(ActionEvent ev){
260         String actionCmd = ev.getActionCommand();
261         return parseActionCommand(actionCmd);
262     }
263 
264     /**
265      * Parse action command to milli sec interval value.
266      *
267      * @param actionCmd command
268      * @return milli sec interval
269      */
270     private int parseActionCommand(String actionCmd){
271         Integer iVal;
272         iVal = this.msecCommand.get(actionCmd);
273         if(iVal == null) return 0;
274         return iVal;
275     }
276 
277     /**
278      * Return Swing timer.
279      *
280      * @return timer
281      */
282     Timer getTimer(){
283         return this.timer;
284     }

287     /**
288      * Interval information.
289      */
290     static class IntervalTick{
291         final int msec;
292         final String label;
293         final boolean isDefault;
294 
295         /**
296          * Constructor.
297          *
298          * @param msec milli second
299          * @param label radio button label
300          */
301         public IntervalTick(int msec, String label){
302             this(msec, label, false);
303             return;
304         }
305 
306         /**
307          * Constructor.
308          *
309          * @param msec milli second
310          * @param label radio button label
311          * @param isDefault true if default radio button
312          */
313         public IntervalTick(int msec, String label, boolean isDefault){
314             super();
315             this.label = label.intern();
316             this.msec = msec;
317             this.isDefault = isDefault;
318             return;
319         }
320 
321     }
